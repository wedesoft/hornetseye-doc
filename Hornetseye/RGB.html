def ==(other)
  if other.is_a? RGB
    @r.eq( other.r ).and( @g.eq( other.g ) ).and( @b.eq( other.b ) )
  elsif RGB.generic? other
    @r.eq(other).and( @g.eq(other) ).and( @b.eq(other) )
  else
    false
  end
end
