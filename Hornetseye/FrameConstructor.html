def method_missing( name, *args )
  if name.to_s =~ /^[a-z0-9]+$/
    target = name.to_s.upcase
    if Hornetseye.const_defined? target
      target = Hornetseye.const_get target
      if target.is_a? FourCC
        new target, *args
      else
        super name, *args
      end
    else
      super name, *args
    end
  else
    super name, *args
  end
end
